# this class contains a IO async request
# can handle thousands of requests concurrently without breaking or using barely and CPU

class AsyncRequests:
    @staticmethod
    def html_get(urls):
        async def fetch(url, session):
            async with session.get(url) as response:
                return await response.read()
    
        async def cooler_get(u):
            tasks = []

            # Fetch all responses within one Client session,
            # keep connection alive for all requests.
            async with ClientSession() as session:
                for i in u:
                    task = asyncio.ensure_future(fetch(i, session))
                    tasks.append(task)

                responses = await asyncio.gather(*tasks)
                # you now have all response bodies in this variable
                return responses

        loop = asyncio.get_event_loop()
        future = asyncio.ensure_future(cooler_get(urls))
        responses = loop.run_until_complete(future)
        return responses

class Month:
    def __init__(self, year: int, month: int):
        self.NUM_DAYS = calendar.monthrange(year, month)
        self.FIRST_DAY = date(year, month, 1)
        self.LAST_DAY = date(year, month, self.NUM_DAYS[1])
        self.PERIODS = []
        for i in range(self.FIRST_DAY.day, self.LAST_DAY.day, 7):
            if not (i + 6 > self.LAST_DAY.day):
                self.PERIODS.append(date(year, month, i))

        self.PERIODS.append(date(year, month, self.LAST_DAY.day - self.LAST_DAY.day % 7))

    @staticmethod
    def df_transform(results):
        soups = [BeautifulSoup(result, 'html.parser') for result in results]
        tables = [soup.find_all('table', class_='list hours responsive') for soup in soups]
        tables = [pd.read_html(str(table[1]), header=0, encoding='utf8')[0] for table in tables]
        tables = [table.drop(['Unnamed: 0', 'Unnamed: 1'], axis=1) for table in tables]
        working_dataframe = pd.concat([i for i in tables], axis=1)
        return working_dataframe

    @staticmethod
    def get_week(dates):
        dates = [i + timedelta(days=6) for i in dates]
        urls = ['https://www.epexspot.com/en/market-data/dayaheadauction/auction-table/'
                + str(date) + '/de' for date in dates]
        results = AsyncRequests.html_get(urls)
        return results
    
    def get_prices(self):
        dates = [i for i in self.PERIODS]
        df = Month.df_transform(Month.get_week(dates))

        index = pd.date_range(start=str(self.FIRST_DAY)
                                    + '-00:00', end=str(self.PERIODS[-1]
                                                        + timedelta(days=7))
                                                    + '-23:00', freq='h')

        prices = df[::2].transpose().stack()
        prices.index = index
        removal = -24 * (self.LAST_DAY.day % 7 + 3) #TODO figure out why this works
        prices = prices[0:removal]

        volumes = df[1::2].transpose().stack()
        volumes.index = index
        volumes = volumes[0:removal]

        df = pd.concat([prices, volumes], axis=1)
        df.columns = ["price", "volume"]
        return df
