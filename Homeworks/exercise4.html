# k is the size of each fold. It is computed dividing the number of 
# rows in X_h (past data) by number_folds. This number is floored and coerced to int
from sklearn.model_selection import TimeSeriesSplit
class TimeSeries:
    
    @staticmethod
    def TimeSeriesCrossValidation(x,n): 
        #numberof_folds = 1000
        h = int(np.floor(float(x.shape[0])))
        k = int(h / n)
        result_array = np.array([])
        for i in range(1, k):   
            observation = -k -h + i
            X_historical = data.drop('pm2_5', axis=1).ix[:observation]
            y_historical = data.pm2_5.ix[:observation]     
            X_forecast = data.drop('pm2_5', axis=1).ix[observation + 1:]
            Y_forecast = data.pm2_5.ix[observation + 1:]           
            reg = RandomForestRegressor(n_estimators=100, oob_score=True)
            reg.fit(X_historical, y_historical)
            reg.predict(X_forecast)
            result_array = np.append(result_array, reg.oob_score_)    
        return np.average(result_array) 
    
    @staticmethod
    def TimeSeriesCrossValidateScore(x,y,reg,n):
        result_array = np.array([])
        tscv = TimeSeriesSplit(n_splits=n)
        for train_index, test_index in tscv.split(x):
            X_train, X_test = x[train_index], x[test_index]
            y_train, y_test = y[train_index], y[test_index]
            reg.fit(X_train, y_train)
            predict = reg.predict(X_test)
            result_array = np.append(result_array,predict)                
        return result_array
